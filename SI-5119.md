    def mapSeparate[VL[_], VR[_]](f: V[_] => ({type l[T] = Either[VL[T], VR[T]]})#l[_] ) = {

        (f(v) match {
	case Left(l) => Left((k, l))
	case Right(r) => Right((k, r))
