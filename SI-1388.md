trait FunctorW[F[_], A] {
  val v: F[A]
  val functor: Functor[F]

  def >[B](f: A => B) = functor.fmap(f, v)
}

object FunctorW {
  def functor[F[_]] = new {
    def apply[A](ft: F[A])(implicit f: Functor[F]) = new FunctorW[F, A] {
      val v = ft
      val functor = f
    }
  }

  implicit def OptionFunctor[A](as: Option[A]) = functor[Option](as)
}
