I wrote a function calculating variance:

{code:scala}
  @inline def variance(data: Array[Double]): Double = {
    val count = data.length
    val mean = data.sum / count
    val devs = data.map(score => (score - mean) * (score - mean))
    devs.sum / count
  }
{code}
