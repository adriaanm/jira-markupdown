    {
        val firstBlock = new Array[Byte](A.BLOCK_SIZE)
        val secondBlock = new Array[Byte](A.BLOCK_SIZE)


        readFile(first) { firstStream =>
            readFile(second) { secondStream =>

                var firstCount = 0
                var secondCount = 0
