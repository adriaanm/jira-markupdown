The following code works fine:

    class Foo[a](val arg : a)

    object Foo  {
    def apply [a](arg : a, right :a) = new Foo[a](arg)
    def unapply [a](m : Foo[a]) = Some (m.arg)
    }
   
    def matchAndGetArgFromFoo[a]( e:Foo[a]):a = {e match { case Foo(x) => x }}
