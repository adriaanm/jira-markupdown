{code}
package bizz

trait DecXTMaker[DTM <: DecXT[DTM]] {
def mkA(ctorData: Long): DTM;
val hashPrecision: Int
}
trait DecXTFactory[DTF <: DecXT[DTF]] extends DecXTMaker[DTF]
