object IMainSpecs extends Specification {

      settings.processArgumentString("-usejavacp")

      class MyIMain extends IMain(settings) {
        def lastRequest = prevRequestList.last
      }

      val n = new MyIMain()
      n.interpret("val a = 1")

      val r = n.lastRequest
      r.lineRep.call("$result").must_== (1)
    }
  }
}
