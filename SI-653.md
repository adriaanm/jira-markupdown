{code}
abstract class Sum[A, B]
{  
}

class InL[A, B](x : A) extends Sum[A, B]
{
   val a : A = x;
}

class InR[A, B](x : B) extends Sum[A, B]
{
   val b : B = x;
}

{
   val roll : Op[Fix[Op]] = x;

class FixTest {
   type Foo[A] = Sum[Unit, A];
   type Nat = Fix[Foo];
   
   val zero = new Fix(new InL(()));
